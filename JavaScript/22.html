	let oTxt=document.getElementById('txt1');
	let oBtn=document.getElementById('btn1');

	oBtn.onclick=function (){
		let re=/^(\d|[1-9]\d{1,3})$/;

		if(re.test(oTxt.value)){
			alert('通过');
		}else{
			alert('不对');
		}
	};
};
